```python
class Solution:
    def containsNearbyDuplicate(self, nums: List[int], k: int) -> bool:
        if len(nums) == 1:
            return False
        i = 0
        d = dict()
        for j in range(len(nums)):
            if d.get(nums[j], 0):
                return True
            d[nums[j]] = 1
            if j - i >= k:
                d[nums[i]] = 0
                i += 1
        return False
```
