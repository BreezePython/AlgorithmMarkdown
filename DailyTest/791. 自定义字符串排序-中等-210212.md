```python
class Solution:
    def customSortString(self, S: str, T: str) -> str:
        if not S:
            return T
        dict_t = Counter(T)
        res = ''
        for s in S:
            temp = dict_t.get(s, 0)
            if temp:
                res += s * temp
            dict_t[s] = 0
        for k, v in dict_t.items():
            if v:
                res += k * v
        return res
```
